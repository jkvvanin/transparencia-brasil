function buscar(tipo) { const nome = document.getElementById('nome').value; const cargo = document.getElementById('cargo').value; const cidade = document.getElementById('cidade').value; const estado = document.getElementById('estado').value; let busca = ""; if (tipo === 'verba') { busca = `${nome} ${cargo} ${cidade} ${estado} "verba de gabinete" OR "gastos" site:gov.br OR site:leg.br`; } else if (tipo === 'salario') { busca = `${nome} ${cargo} ${cidade} ${estado} "subsídio" OR "remuneração" site:gov.br OR site:leg.br`; } else if (tipo === 'patrimonio') { // ESTE É O COMANDO QUE BUSCA OS BENS NO TSE busca = `${nome} ${cargo} ${cidade} ${estado} "declaração de bens" OR "patrimônio" TSE`; } window.open(`https://www.google.com/search?q=${encodeURIComponent(busca)}`, '_blank'); }
